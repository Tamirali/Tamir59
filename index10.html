// api/generate.js (Backend Node.js API) export default async function handler(req, res) { // CORS Headers अनुमति देने के लिए res.setHeader('Access-Control-Allow-Origin', '*'); res.setHeader('Access-Control-Allow-Methods', 'POST, OPTIONS'); res.setHeader('Access-Control-Allow-Headers', 'Content-Type'); if (req.method === 'OPTIONS') { return res.status(200).end(); } const { photoUrl, audioUrl } = req.body; try { const response = await fetch('https://api.replicate.com/v1/predictions', { method: 'POST', headers: { 'Authorization': `Token ${process.env.REPLICATE_API_KEY}`, // Backend में API Key सुरक्षित रखें 'Content-Type': 'application/json' }, body: JSON.stringify({ version: "28e0856096531398188168234390f701c640e7025be452380775d7eef84ff216", input: { source_image: photoUrl, driven_audio: audioUrl } }) }); const data = await response.json(); return res.status(200).json(data); } catch (error) { return res.status(500).json({ error: error.message }); } }
